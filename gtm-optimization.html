Option 1: Lazy Load GTM HIGH IMPACT

Delay loading GTM until after the page is interactive. This improves initial load time significantly.

Replace your current GTM script with:

// In index.html - REMOVE the existing async GTM script
// <script async src="https://www.googletagmanager.com/gtag/js?id=G-GVGKYFGZ1Z"></script>

// Add this instead in your <head>:
<script>
// Lazy load GTM after page is interactive
window.addEventListener('load', function() {
    setTimeout(function() {
        // Create script element
        const script = document.createElement('script');
        script.async = true;
        script.src = 'https://www.googletagmanager.com/gtag/js?id=G-GVGKYFGZ1Z';

        // Add the script
        document.head.appendChild(script);

        // Initialize after script loads
        script.onload = function() {
            window.dataLayer = window.dataLayer || [];
            function gtag(){dataLayer.push(arguments);}
            gtag('js', new Date());
            gtag('config', 'G-GVGKYFGZ1Z');
        };
    }, 1000); // Delay by 1 second after page load
});
</script>

Option 3: Minimal GA4 Implementation HIGH IMPACT

Replace GTM with a minimal GA4 implementation for basic tracking.

Ultra-light GA4 tracking (only 2KB!):

// Minimal GA4 implementation - add to your app.js
const GA_MEASUREMENT_ID = 'G-GVGKYFGZ1Z';

// Simple pageview tracking
function trackPageview() {
    if (navigator.sendBeacon) {
        const url = `https://www.google-analytics.com/g/collect?v=2&tid=${GA_MEASUREMENT_ID}&sid=${Date.now()}&_p=${Date.now()}&sr=${screen.width}x${screen.height}&ul=${navigator.language}&cid=${localStorage.getItem('ga_client_id') || Math.random()}&dl=${encodeURIComponent(location.href)}&dt=${encodeURIComponent(document.title)}&en=page_view`;
        navigator.sendBeacon(url);
    }
}

// Simple event tracking
function trackEvent(eventName, parameters = {}) {
    if (navigator.sendBeacon) {
        const params = new URLSearchParams({
            v: 2,
            tid: GA_MEASUREMENT_ID,
            sid: Date.now(),
            _p: Date.now(),
            cid: localStorage.getItem('ga_client_id') || Math.random(),
            en: eventName,
            ...parameters
        });
        navigator.sendBeacon(`https://www.google-analytics.com/g/collect?${params}`);
    }
}

// Track initial pageview
trackPageview();

// Track loadout generation
document.getElementById('generateBtn')?.addEventListener('click', () => {
    trackEvent('generate_loadout', {
        ep_class: selectedClass,
        ep_spins: selectedSpins
    });
});

Option 4: Load GTM Only for Returning Users LOW IMPACT

First-time visitors get fast load, returning users get full analytics.

// Check if returning user
const isReturningUser = localStorage.getItem('returning_user');

if (isReturningUser) {
    // Load GTM normally for returning users
    const script = document.createElement('script');
    script.async = true;
    script.src = 'https://www.googletagmanager.com/gtag/js?id=G-GVGKYFGZ1Z';
    document.head.appendChild(script);
} else {
    // First visit - set flag for next time
    localStorage.setItem('returning_user', 'true');

    // Use minimal tracking for first visit
    navigator.sendBeacon('https://www.google-analytics.com/g/collect?v=2&tid=G-GVGKYFGZ1Z&en=first_visit');
}
